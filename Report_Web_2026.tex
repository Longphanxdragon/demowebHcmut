\section{T\'inh n\u{a}ng hi\d{e}n th\d{u}c (b\'am s\'at \`e)}
\subsection{Kh\'ach}
\begin{itemize}[leftmargin=*]
  \item Xem c\'ac trang public: trang ch\u{u}, gi\h{o}i thi\d{e}u, s\h{a}n ph\h{a}m, b\h{a}ng gi\'a, tin t\u{u}c, li\^en h\d{e}, FAQ.
  \item T\`im ki\'em s\h{a}n ph\h{a}m theo t\`u kh\'oa.
  \item T\`im ki\'em tin t\u{u}c theo t\`u kh\'oa.
  \item Xem trang chi ti\'et s\h{a}n ph\h{a}m và chi ti\'et tin t\u{u}c.
  \item G\h{u}i form li\^en h\d{e} (c\'o validate client/server).
\end{itemize}

\subsection{Th\`anh vi\^en}
\begin{itemize}[leftmargin=*]
  \item \DJ \u{a}ng k\'y / \DJ \u{a}ng nh\d{a}p.
  \item C\d{a}p nh\d{a}t th\^ong tin c\'a nh\^an (t\^en, avatar).
  \item \DJ \h{o}i m\d{a}t kh\h{a}u (ki\h{e}m tra m\d{a}t kh\h{a}u c\~u, x\'ac nh\d{a}n m\d{a}t kh\h{a}u m\'oi).
\end{itemize}

\subsection{Qu\h{a}n tr\d{i} vi\^en}
\begin{itemize}[leftmargin=*]
  \item Qu\h{a}n l\'y s\h{a}n ph\h{a}m: th\^em/x\'oa (AJAX).
  \item Qu\h{a}n l\'y th\`anh vi\^en: xem danh s\'ach, kho\'a/m\h{o} kho\'a, reset m\d{a}t kh\h{a}u, x\'oa (kh\^ong \h{a}nh h\h{u}\h{o}ng t\`ai kho\h{a}n admin). 
  \item Qu\h{a}n l\'y li\^en h\d{e}: c\d{a}p nh\d{a}t tr\d{a}ng th\'ai new/read/replied.
  \item Ph\^an trang danh s\'ach li\^en h\d{e} tr\^en trang admin.
  \item Ki\h{e}m tra d\~u li\d{e}u action admin \h{o} m\d{u}c c\u{o} b\h{a}n (ID h\h{o}p l\d{e}, tr\h{a}ng th\'ai h\h{o}p l\d{e}, m\d{a}t kh\h{a}u m\'oi \ge 6 k\'y t\d{u}).
\end{itemize}

\subsection{Ph\^an trang}
\begin{itemize}[leftmargin=*]
  \item S\h{a}n ph\h{a}m: ph\^an trang theo \texttt{product\_page}.
  \item Tin t\u{u}c: ph\^an trang theo \texttt{news\_page}.
  \item Li\^en h\d{e} admin: ph\^an trang theo \texttt{contact\_page}.
\end{itemize}

\section{B\h{a}o m\d{a}t c\u{o} b\h{a}n v\`a SEO}
\subsection{B\h{a}o m\d{a}t}
\begin{itemize}[leftmargin=*]
  \item M\d{a}t kh\h{a}u d\`ung \texttt{password\_hash}/\texttt{password\_verify}.
  \item Escape output b\`ang \texttt{htmlspecialchars} \d{e} gi\h{a}m XSS.
  \item D\`ung prepared statement khi truy v\'an CSDL.
  \item Ph\^an quy\`en theo role cho trang admin.
  \item Ch\h{a}n thao t\'ac nguy hi\h{e}m: kh\^ong cho admin t\d{u} kho\'a/x\'oa ch\'inh t\`ai kho\h{a}n \d{a}ng \d{d}\u{a}ng nh\d{a}p.
\end{itemize}

\subsection{SEO c\u{o} b\h{a}n}
\begin{itemize}[leftmargin=*]
  \item Thi\'et l\d{a}p \texttt{title} theo t\`ung trang.
  \item C\'au tr\'uc n\d{o}i dung c\'o heading r\~o r\`ang.
  \item URL c\'o tham s\'o trang/t\`im ki\'em ph\^an bi\d{e}t theo ch\u{u}c n\u{a}ng.
\end{itemize}

\section{Ki\h{e}m th\h{u} v\`a m\^oi tr\u{o}\`ng ch\d{a}y}
\subsection{M\^oi tr\u{o}\`ng}
\begin{itemize}[leftmargin=*]
  \item PHP \ge 7.0 (khuy\'en ngh\d{i} PHP 8.x).
  \item Web server: Apache/Nginx (ho\d{a}c \texttt{php -S}).
  \item CSDL: SQLite file ho\d{a}c MySQL (theo script tham chi\'eu).
\end{itemize}

\subsection{C\'ach ch\d{a}y nhanh}
\begin{enumerate}[leftmargin=*]

  \item Ch\d{a}y \texttt{php -S localhost:8000}.
  \item M\h{o} \texttt{/index.php} tr\^en tr\`inh duy\d{e}t.
  \item T\`ai kho\h{a}n admin m\~au: \texttt{admin@vng.com}.
\end{enumerate}
